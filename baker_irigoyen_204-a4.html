Question 3

discounts <- c(.01, .02, .03, .04, .05)
scc <- tribble(~c1, ~c2)
for (i in discounts){
  disc <- i
  npv <- discount(war$diff_ton, 
                  disc,
                  calculate_NPV = TRUE)
  bri <- c(disc, npv)
  scc <- rbind(scc, bri)
} 
scc <- scc %>% 
  rename("Discount_Rate" = "X0.01",
         "SCC" = "X183.316049160008")
ggplot(scc)+
  geom_line(aes(x = Discount_Rate, y = SCC))+
  theme_minimal()
